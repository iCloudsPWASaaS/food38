public function getCoverAttribute(): string
{
    $media = $this->getMedia('item')->last();

    if (!$media) {
        return ''; // No media attached
    }

    // Get full URL or path of the cover conversion
    $path = $media->getPath(); // local storage path
    // or use getUrl('cover') if you want URL

    // Extract file name only
    return basename($path);
}

jsholdOrder: function () {
    if (this.carts.length === 0) {
        return alertService.error("Cart empty");
    }
    
    // Create a unique ID for this held order
    const orderId = 'hold_' + Date.now();
    
    // Save the complete order details
    const heldOrder = {
        id: orderId,
        timestamp: Date.now(),
        items: JSON.parse(JSON.stringify(this.carts)), // Deep copy of cart items with all details
        customer: this.checkoutProps.form.customer_id,
        customerName: this.customers.find(c => c.id === this.checkoutProps.form.customer_id)?.name || '',
        token: this.checkoutProps.form.token,
        table: this.checkoutProps.form.dining_table_id,
        tableName: this.diningtables.find(t => t.id === this.checkoutProps.form.dining_table_id)?.name || '',
        orderType: this.checkoutProps.form.order_type,
        discount: this.discount,
        discountType: this.discountType,
        address: this.checkoutProps.form.address_id,
        subtotal: this.subtotal,
        deliveryCharge: this.checkoutProps.form.delivery_charge
    };
    
    // Get existing held orders or initialize empty array
    let existingOrders = [];
    try {
        const savedOrders = localStorage.getItem('foodking_held_orders');
        if (savedOrders) {
            existingOrders = JSON.parse(savedOrders);
        }
    } catch (e) {
        console.error('Error loading held orders', e);
    }
    
    // Add new held order
    existingOrders.push(heldOrder);
    
    // Save back to localStorage
    localStorage.setItem('foodking_held_orders', JSON.stringify(existingOrders));
    
    // Reset the cart
    this.resetCart();
    
    alertService.success("Order hold successfully");
},

openHoldOrdersModal: function () {
    // Load held orders from localStorage
    this.loadHeldOrders();
    
    // Filter to show only today's orders
    this.filterTodayOrders();
    
    // Open the modal
    appService.modalShow('#holdOrdersModal');
},

loadHeldOrders: function () {
    try {
        const savedOrders = localStorage.getItem('foodking_held_orders');
        if (savedOrders) {
            this.holdOrders = JSON.parse(savedOrders);
        } else {
            //this.holdOrders = [];
        }
    } catch (e) {
        console.error('Error loading held orders', e);
        this.holdOrders = [];
    }
},

filterTodayOrders: function () {
    // Get today's date at midnight for comparison
    const today = new Date();
    today.setHours(0, 0, 0, 0);
    
    // Filter orders from today only
    this.holdOrders = this.holdOrders.filter(order => {
        const orderDate = new Date(order.timestamp);
        return orderDate >= today;
    });
    
    // Save filtered list back to localStorage
    localStorage.setItem('foodking_held_orders', JSON.stringify(this.holdOrders));
},

loadHoldOrder: function (orderId) {
    // Find the order in the held orders list
    const orderIndex = this.holdOrders.findIndex(order => order.id === orderId);
    if (orderIndex === -1) return;
    
    const order = this.holdOrders[orderIndex];

    // Store the current hold order ID
    this.currentHoldOrderId = orderId;
    
    // First reset the current cart
    this.$store.dispatch('posCart/resetCart').then(() => {
        // Restore cart items with all details
        this.$store.dispatch('posCart/lists', order.items).then(() => {
            // Restore customer, token, table, etc.
            this.checkoutProps.form.customer_id = order.customer;
            this.checkoutProps.form.token = order.token;
            this.checkoutProps.form.dining_table_id = order.table;
            this.checkoutProps.form.order_type = order.orderType;
            this.checkoutProps.form.address_id = order.address;
            this.checkoutProps.form.delivery_charge = order.deliveryCharge;
            
            // Restore discount
            this.discount = order.discount;
            this.discountType = order.discountType;
            if (order.discount > 0) {
                this.applyDiscount();
            }
            
            // If order has a customer, load their addresses
            if (order.customer) {
                this.gettingUserAddress(order.customer);
            }
            
            // Remove the order from held orders
            //this.holdOrders.splice(orderIndex, 1);
            //localStorage.setItem('foodking_held_orders', JSON.stringify(this.holdOrders));
            
            // Close the modal
            appService.modalHide('#holdOrdersModal');
            
            alertService.success("Order loaded successfully");
        });
    });
},

deleteHoldOrder: function (orderId) {
    // Find and remove the order
    const orderIndex = this.holdOrders.findIndex(order => order.id === orderId);
    if (orderIndex !== -1) {
        this.holdOrders.splice(orderIndex, 1);
        localStorage.setItem('foodking_held_orders', JSON.stringify(this.holdOrders));
        alertService.success(this.$t('message.order_deleted_successfully'));
    }
},

formatTime: function (timestamp) {
    const date = new Date(timestamp);
    return date.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
},

closeHoldOrdersModal: function() {
    appService.modalHide('#holdOrdersModal');
},
